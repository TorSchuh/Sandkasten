27      public static void main(final String[] args) {
28          
29          Gameplay gameplay = new Gameplay(); 
30          PlayboardTicTacToe playboard = new PlayboardTicTacToe();
31          String humanTokenTypeSymbol = askForHumanTokenType();
32          boolean endOfGame = false;
33                  
34          do {
35              if (!gameplay.isComputerNext()) {
36                  playboard.printPlayboard();
37              }   
38              if (gameplay.isComputerNext()) {
39                 if (gameplay.moveComputer(playboard, getComputerToken(humanTokenTypeSymbol))) {
40                     gameplay.setComputerNext(false);
41                 }
42              } else if (!gameplay.isComputerNext()) {
43                  if (gameplay.moveHuman(playboard, getHumanToken(humanTokenTypeSymbol), askForHumanNextMove(gameplay, playboard))) {
44                      gameplay.setComputerNext(true);
45                  }
46              }
47              if (gameplay.isFinishedGame()) {
48                  if (gameplay.isComputerNext()) {
49                      System.out.println("\n VICTORY :)");
50                  } else {
51                      System.out.println("\n GAME OVER :(");
52                  }
53                  endOfGame = !askForRematch();
54                  if (!endOfGame) {
55                      gameplay = new Gameplay(); 
56                      playboard = new PlayboardTicTacToe();
57                      humanTokenTypeSymbol = askForHumanTokenType();
58                  }
59              }
60          } while (!endOfGame);
61      }

63      private static boolean askForRematch() {
64          BufferedReader input = new BufferedReader(new InputStreamReader(System.in));
65          boolean inputAccepted = false;
66          String decicion = "";
67          try {
68              do {
69                  System.out.println("\n do you want a rematch [y] or [n]? : ");
70                  decicion = input.readLine().trim().toUpperCase();    
71                  if (decicion.equals("Y") || decicion.equals("N")) {
72                      inputAccepted = true;
73                  }
74              } while (!inputAccepted);
75          } catch (IOException e) {
76              System.out.println("Error reading user input!");
77          }
78          return decicion.equals("Y");
79      }

88                  symbol = input.readLine().trim().toUpperCase();    
89                  if (symbol.equals(AbstractToken.TOKEN_CIRCLE) || symbol.equals(AbstractToken.TOKEN_CROSS)) {
90                      symbolAccepted = true;
91                  }
92              } while (!symbolAccepted);
93          } catch (IOException e) {
94              System.out.println("Error reading user input!");

99      private static Identifier askForHumanNextMove(Gameplay pGameplay, PlayboardTicTacToe pPlayboard) {
100         BufferedReader input = new BufferedReader(new InputStreamReader(System.in));
101         boolean identifierAccepted = false;
102         String identifier = ""; 
103         try {
104             do {
105                 System.out.println("\n please put in your next move: ");
106                 identifier = input.readLine().trim().toUpperCase();    
107                 if (pGameplay.isValidMove(pPlayboard, Identifier.valueOf(identifier))) {
108                     identifierAccepted = true;
109                 }
110             } while (!identifierAccepted);
111         } catch (IOException e) {
112             System.out.println("Error reading user input!");
113         }

28      public boolean moveComputer(PlayboardTicTacToe pPlayboard, AbstractToken pToken) {      
29          
30          // test if computer can win
31          for (Field field : pPlayboard.getFields()) {
32              AbstractField fieldTmp = (AbstractField) field;
33              Token tokenTmp = fieldTmp.getToken();
34              if (isValidMove(pPlayboard, fieldTmp.getIdentifier())) {
35                  fieldTmp.setToken(pToken);
36                  if (isFinishingMove(pPlayboard, ((AbstractField) field).getIdentifier(), pToken)) {
37                      this.isFinishedGame = true;
38                      return true;
39                  }
40              }
41              fieldTmp.setToken(tokenTmp);
42          }
43          
44          // test if human can win
45          for (Field field : pPlayboard.getFields()) {
46              AbstractField fieldTmp = (AbstractField) field;
47              Token tokenTmp = fieldTmp.getToken();
48              AbstractToken tokenOpponent = new TokenCircle();
49              if (pToken.getType().equals(TokenType.Type.CIRCLE)) {
50                  tokenOpponent = new TokenCross();
51              }
52              if (isValidMove(pPlayboard, fieldTmp.getIdentifier())) {
53                  fieldTmp.setToken(tokenOpponent);
54                  if (isFinishingMove(pPlayboard, ((AbstractField) field).getIdentifier(), tokenOpponent)) {
55                      return true;
56                  }
57              }
58              fieldTmp.setToken(tokenTmp);
59          }
60          
61          // test if center field is empty
62          if (((AbstractToken) pPlayboard.getField(Identifier.B2).getToken()).getType().equals(TokenType.Type.BLANK)) {
63              pPlayboard.getField(Identifier.B2).setToken(pToken);
64              return true;
65          }
66          
67          // choose any/next free field
68          for (Field field : pPlayboard.getFields()) {
69              AbstractField fieldTmp = (AbstractField) field;
70              Token tokenTmp = fieldTmp.getToken();
71              if (isValidMove(pPlayboard, fieldTmp.getIdentifier())) {
72                  fieldTmp.setToken(pToken);
73                  pPlayboard.getField(fieldTmp.getIdentifier()).setToken(pToken);
74                  return true;
75              }
76              fieldTmp.setToken(tokenTmp);
77          }
78          
79          // no empty field
80          return false;
81      }

83      public boolean moveHuman(PlayboardTicTacToe pPlayboard, AbstractToken pToken, Identifier pIdentifier) {
84          if (isValidMove(pPlayboard, pIdentifier)) {
85              pPlayboard.getField(pIdentifier).setToken(pToken);
86              if (isFinishingMove(pPlayboard, pIdentifier, pToken)) {
87                  this.isFinishedGame = true;

89              return true;
90          }
91          return false;

98      private boolean isFinishingMove(final PlayboardTicTacToe pPlayboard, final Identifier pIdentifier, final AbstractToken pToken) {
99          return Rules.isSameToken_A1_A2_A3(pPlayboard, pToken.getType()) || Rules.isSameToken_A1_B1_C1(pPlayboard, pToken.getType())
100                 || Rules.isSameToken_A1_B2_C3(pPlayboard, pToken.getType()) || Rules.isSameToken_A2_B2_C2(pPlayboard, pToken.getType())
101                 || Rules.isSameToken_A3_B3_C3(pPlayboard, pToken.getType()) || Rules.isSameToken_B1_B2_B3(pPlayboard, pToken.getType())
102                 || Rules.isSameToken_C1_B2_A3(pPlayboard, pToken.getType()) || Rules.isSameToken_C1_C2_C3(pPlayboard, pToken.getType());
103     }
